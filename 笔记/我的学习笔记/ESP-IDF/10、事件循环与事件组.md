# ESP-IDF 事件循环与事件组 （事件驱动架构与任务同步）

## 核心概念

- **事件循环** - ESP-IDF的核心机制，用于分发和处理系统事件
- **系统事件** - WiFi、IP等底层事件通过事件循环分发
- **自定义事件** - 用户可定义自己的事件类型
- **事件组** - 用于多任务同步和状态管理的位标志

---

## 一、默认事件循环

### 1.1 创建默认事件循环

```c
#include "esp_event.h"

// 创建默认事件循环（只能创建一个）
ESP_ERROR_CHECK(esp_event_loop_create_default());

// 此函数内部创建了一个名为"default"的事件循环
// 大多数ESP-IDF组件（WiFi、ETH等）使用这个默认循环
```

**事件循环架构：**

```
事件源 ──> 事件投递 ──> 事件循环 ──> 事件分发 ──> 事件处理器
   │                        │                        │
   │                        │                        │
   ▼                        ▼                        ▼
WiFi事件               事件队列               注册的处理函数
IP事件                 (FIFO)                  用户回调
自定义事件                                      任务通知
```

---

### 1.2 注册事件处理器

```c
// 定义事件处理器
void wifi_event_handler(void *handler_args, 
                        esp_event_base_t base, 
                        int32_t id, 
                        void *event_data)
{
    if (base == WIFI_EVENT) {
        switch (id) {
            case WIFI_EVENT_STA_START:
                ESP_LOGI(TAG, "WiFi started");
                break;
            case WIFI_EVENT_STA_CONNECTED:
                ESP_LOGI(TAG, "WiFi connected");
                break;
            case WIFI_EVENT_STA_DISCONNECTED:
                ESP_LOGI(TAG, "WiFi disconnected");
                break;
        }
    }
}

// 注册到默认事件循环
ESP_ERROR_CHECK(esp_event_handler_instance_register(
    WIFI_EVENT,           // 事件基
    ESP_EVENT_ANY_ID,     // 事件ID（ANY=所有）
    &wifi_event_handler,  // 处理函数
    NULL,                 // 传递给处理器的参数
    NULL                  // 实例句柄（可选）
));
```

| 参数 | 说明 | 示例 |
|------|------|------|
| `event_base` | 事件基标识 | `WIFI_EVENT`, `IP_EVENT` |
| `event_id` | 事件ID | `WIFI_EVENT_STA_START` |
| `event_handler` | 处理函数 | 用户定义的回调 |
| `handler_args` | 处理器参数 | 任意指针 |

---

### 1.3 注销事件处理器

```c
// 注销特定处理器
ESP_ERROR_CHECK(esp_event_handler_instance_unregister(
    WIFI_EVENT,
    ESP_EVENT_ANY_ID,
    instance           // 注册时获得的实例句柄
));

// 注销所有WIFI_EVENT处理器
ESP_ERROR_CHECK(esp_event_handler_unregister(WIFI_EVENT, ESP_EVENT_ANY_ID, &wifi_event_handler));
```

### 2.1 创建自定义事件循环

```c
// 事件循环句柄
esp_event_loop_handle_t my_loop_handle;

// 事件循环配置
esp_event_loop_args_t loop_args = {
    .queue_size = 10,           // 事件队列大小
    .task_name = "my_evt_loop", // 任务名称

ESP_ERROR_CHECK(esp_event_loop_create(&loop_args, &my_loop_handle));
```

## 三、事件定义与发送

### 3.1 定义自定义事件基

```c
// 在头文件中声明
ESP_EVENT_DECLARE_BASE(MY_EVENT_BASE);

// 在源文件中定义
ESP_EVENT_DEFINE_BASE(MY_EVENT_BASE);

// 定义事件ID枚举
enum {
    MY_EVENT_ID_1,      // 0
    MY_EVENT_ID_2,      // 1
    MY_EVENT_ID_3,      // 2
    MY_EVENT_COUNT
};
```

---

### 3.2 发送事件

```c
// 1. 发送简单事件（无数据）
ESP_ERROR_CHECK(esp_event_post(
    MY_EVENT_BASE,       // 事件基
    MY_EVENT_ID_1,       // 事件ID
    NULL,                // 事件数据
    0,                   // 数据长度
    portMAX_DELAY        // 超时
));

// 2. 发送带数据的事件
typedef struct {
    int sensor_id;
    float value;
    uint64_t timestamp;
} sensor_data_t;

sensor_data_t data = {
    .sensor_id = 1,
    .value = 25.5,
    .timestamp = esp_timer_get_time()
};

ESP_ERROR_CHECK(esp_event_post(
    MY_EVENT_BASE,
    MY_EVENT_ID_2,
    &data,               // 事件数据
    sizeof(data),        // 数据长度
    portMAX_DELAY
));

// 3. 发送到指定事件循环
ESP_ERROR_CHECK(esp_event_post_to(
    my_loop_handle,      // 目标事件循环
    MY_EVENT_BASE,
    MY_EVENT_ID_3,
    &data,
    sizeof(data),
    portMAX_DELAY
));
```

### 3.3 事件处理器示例

```c
void my_event_handler(void *handler_args,
```

### 4.2 设置和清除位

```c
// 定义事件位
#define BIT_0    (1 << 0)    // 0x01
#define BIT_1    (1 << 1)    // 0x02
#define BIT_2    (1 << 2)    // 0x04
#define BIT_ALL  (BIT_0 | BIT_1 | BIT_2)

// 设置位（任务中使用）
xEventGroupSetBits(xEventGroup, BIT_0);

// 设置位（中断中使用）
BaseType_t xHigherPriorityTaskWoken = pdFALSE;
xEventGroupSetBitsFromISR(xEventGroup, BIT_1, &xHigherPriorityTaskWoken);
portYIELD_FROM_ISR(xHigherPriorityTaskWoken);

// 清除位
xEventGroupClearBits(xEventGroup, BIT_0);

// 清除位（中断中）
xEventGroupClearBitsFromISR(xEventGroup, BIT_1);

// 获取当前位值
EventBits_t uxBits = xEventGroupGetBits(xEventGroup);
```

## 五、多任务同步

### 5.1 等待事件位

```c
// 等待单个位
EventBits_t uxBits = xEventGroupWaitBits(
    xEventGroup,           // 事件组
    BIT_0,                 // 要等待的位
    pdTRUE,                // 退出时清除位
    pdFALSE,               // 不等待所有位（任意一个即可）
    portMAX_DELAY          // 无限等待
);

// 等待多个位（全部）
uxBits = xEventGroupWaitBits(
    xEventGroup,
    BIT_0 | BIT_1 | BIT_2, // 等待所有位
    pdTRUE,                // 清除位
    pdTRUE,                // 等待所有位
    pdMS_TO_TICKS(5000)    // 5秒超时
);

if ((uxBits & (BIT_0 | BIT_1 | BIT_2)) == (BIT_0 | BIT_1 | BIT_2)) {
    ESP_LOGI(TAG, "All bits set");
} else {
    ESP_LOGW(TAG, "Timeout waiting for bits");
}
```

| 参数 | 说明 | 值 |
|------|------|-----|
| `xClearOnExit` | 退出时清除 | `pdTRUE`/`pdFALSE` |
| `xWaitForAllBits` | 等待所有位 | `pdTRUE`=`AND`, `pdFALSE`=`OR` |
| `xTicksToWait` | 超时时间 | `portMAX_DELAY`或毫秒 |

---

### 5.2 同步点

```c
// 多任务同步点
#define TASK_1_BIT  (1 << 0)
#define TASK_2_BIT  (1 << 1)
#define TASK_3_BIT  (1 << 2)

// 各任务到达同步点
void task1(void *arg)
{
    // 执行任务1的工作
    do_task1_work();
    
    // 到达同步点
    xEventGroupSync(xEventGroup, TASK_1_BIT, 
                    TASK_1_BIT | TASK_2_BIT | TASK_3_BIT, 
                    portMAX_DELAY);
    
    // 所有任务都到达后，继续执行
    ESP_LOGI(TAG, "Task 1 past sync point");
}

void task2(void *arg)
{
    do_task2_work();
    xEventGroupSync(xEventGroup, TASK_2_BIT,
                    TASK_1_BIT | TASK_2_BIT | TASK_3_BIT,
                    portMAX_DELAY);
    ESP_LOGI(TAG, "Task 2 past sync point");
}

void task3(void *arg)
{
    do_task3_work();
    xEventGroupSync(xEventGroup, TASK_3_BIT,
                    TASK_1_BIT | TASK_2_BIT | TASK_3_BIT,
                    portMAX_DELAY);
    ESP_LOGI(TAG, "Task 3 past sync point");
}
```

**同步点示意：**

```
时间轴: ──────────────────────────────────────────>

Task1: ─────┬──────────────────────────┬─────────>
            │ 工作                     │ 同步后继续
            └──> xEventGroupSync() ────┘

Task2: ───────────┬────────────────────┬─────────>
                  │ 工作               │ 同步后继续
                  └────> xEventGroupSync() ─────┘

Task3: ─────────────────┬──────────────┬─────────>
                        │ 工作         │ 同步后继续
                        └──────> xEventGroupSync() ─┘
                              ↑
                        所有任务到达后同时继续
```

## 七、完整示例

### 示例1：WiFi+HTTP连接状态机

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/event_groups.h"
#include "esp_wifi.h"
#include "esp_event.h"
#include "esp_log.h"
#include "nvs_flash.h"

static const char *TAG = "STATE_MACHINE";

// 事件位定义
#define WIFI_CONNECTED_BIT      BIT0
#define WIFI_FAIL_BIT           BIT1
#define HTTP_CONNECTED_BIT      BIT2
#define HTTP_DATA_RECEIVED_BIT  BIT3

static EventGroupHandle_t s_connect_event_group;

// 状态定义
typedef enum {
    STATE_INIT,
    STATE_WIFI_CONNECTING,
    STATE_WIFI_CONNECTED,
    STATE_HTTP_CONNECTING,
    STATE_HTTP_CONNECTED,
    STATE_RUNNING,
    STATE_ERROR
} system_state_t;

static volatile system_state_t current_state = STATE_INIT;

static void event_handler(void *arg, esp_event_base_t event_base,
                          int32_t event_id, void *event_data)
{
    if (event_base == WIFI_EVENT) {
        if (event_id == WIFI_EVENT_STA_START) {
            current_state = STATE_WIFI_CONNECTING;
            esp_wifi_connect();
        } else if (event_id == WIFI_EVENT_STA_CONNECTED) {
            current_state = STATE_WIFI_CONNECTED;
            xEventGroupSetBits(s_connect_event_group, WIFI_CONNECTED_BIT);
        } else if (event_id == WIFI_EVENT_STA_DISCONNECTED) {
            xEventGroupSetBits(s_connect_event_group, WIFI_FAIL_BIT);
            current_state = STATE_ERROR;
        }
    } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t *event = (ip_event_got_ip_t *)event_data;
        ESP_LOGI(TAG, "Got IP: " IPSTR, IP2STR(&event->ip_info.ip));
        current_state = STATE_HTTP_CONNECTING;
        // 触发HTTP连接...
        xEventGroupSetBits(s_connect_event_group, HTTP_CONNECTED_BIT);
    }
}

void state_machine_task(void *arg)
{
    while (1) {
        EventBits_t bits = xEventGroupWaitBits(
            s_connect_event_group,
            WIFI_CONNECTED_BIT | HTTP_CONNECTED_BIT | HTTP_DATA_RECEIVED_BIT,
            pdTRUE, pdFALSE, portMAX_DELAY);
        
        if (bits & WIFI_CONNECTED_BIT) {
            ESP_LOGI(TAG, "State: WiFi connected");
        }
        
        if (bits & HTTP_CONNECTED_BIT) {
            ESP_LOGI(TAG, "State: HTTP connected");
            current_state = STATE_RUNNING;
        }
        
        if (bits & HTTP_DATA_RECEIVED_BIT) {
            ESP_LOGI(TAG, "State: Data received, processing...");
        }
    }
}

void app_main(void)
{
    ESP_ERROR_CHECK(nvs_flash_init());
    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    
    s_connect_event_group = xEventGroupCreate();
    
    esp_netif_create_default_wifi_sta();
    
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&cfg));
    
    ESP_ERROR_CHECK(esp_event_handler_register(
        WIFI_EVENT, ESP_EVENT_ANY_ID, &event_handler, NULL));
    ESP_ERROR_CHECK(esp_event_handler_register(
        IP_EVENT, IP_EVENT_STA_GOT_IP, &event_handler, NULL));
    
    wifi_config_t wifi_config = {
        .sta = {
            .ssid = "YOUR_SSID",
            .password = "YOUR_PASSWORD",
        },
    };
    
    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config));
    ESP_ERROR_CHECK(esp_wifi_start());
    
    xTaskCreate(state_machine_task, "state_machine", 4096, NULL, 5, NULL);
}
```

### 示例2：多传感器数据采集同步

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/event_groups.h"
#include "esp_log.h"

static const char *TAG = "SENSOR_SYNC";

// 传感器就绪位
#define TEMP_READY_BIT      BIT0
#define HUMI_READY_BIT      BIT1
#define PRESS_READY_BIT     BIT2
#define ALL_SENSORS_READY   (TEMP_READY_BIT | HUMI_READY_BIT | PRESS_READY_BIT)

static EventGroupHandle_t sensor_event_group;

// 传感器数据结构
typedef struct {
    float temperature;
    float humidity;
    float pressure;
    uint64_t timestamp;
} sensor_data_t;

static sensor_data_t sensor_data = {0};

// 温度传感器任务
void temp_sensor_task(void *arg)
{
    while (1) {
        // 模拟读取温度
        vTaskDelay(pdMS_TO_TICKS(100));
        sensor_data.temperature = 25.0f + (rand() % 10) / 10.0f;
        
        ESP_LOGI(TAG, "Temperature: %.1f C", sensor_data.temperature);
        xEventGroupSetBits(sensor_event_group, TEMP_READY_BIT);
        
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

// 湿度传感器任务
void humi_sensor_task(void *arg)
{
    while (1) {
        vTaskDelay(pdMS_TO_TICKS(150));
        sensor_data.humidity = 50.0f + (rand() % 20);
        
        ESP_LOGI(TAG, "Humidity: %.1f %%", sensor_data.humidity);
        xEventGroupSetBits(sensor_event_group, HUMI_READY_BIT);
        
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

// 气压传感器任务
void press_sensor_task(void *arg)
{
    while (1) {
        vTaskDelay(pdMS_TO_TICKS(200));
        sensor_data.pressure = 1013.25f + (rand() % 100) / 10.0f;
        
        ESP_LOGI(TAG, "Pressure: %.2f hPa", sensor_data.pressure);
        xEventGroupSetBits(sensor_event_group, PRESS_READY_BIT);
        
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}

// 数据处理任务
void data_process_task(void *arg)
{
    while (1) {
        // 等待所有传感器就绪
        EventBits_t bits = xEventGroupWaitBits(
            sensor_event_group,
            ALL_SENSORS_READY,
            pdTRUE,     // 清除位
            pdTRUE,     // 等待所有位
            pdMS_TO_TICKS(5000)
        );
        
        if ((bits & ALL_SENSORS_READY) == ALL_SENSORS_READY) {
            sensor_data.timestamp = esp_timer_get_time();
            
            ESP_LOGI(TAG, "=== All Sensors Ready ===");
            ESP_LOGI(TAG, "Temp: %.1f C, Humi: %.1f %%, Press: %.2f hPa",
                     sensor_data.temperature,
                     sensor_data.humidity,
                     sensor_data.pressure);
            
            // 发送到云端或存储...
        } else {
            ESP_LOGW(TAG, "Timeout waiting for sensors");
        }
    }
}

void app_main(void)
{
    sensor_event_group = xEventGroupCreate();
    
    xTaskCreate(temp_sensor_task, "temp_task", 2048, NULL, 5, NULL);
    xTaskCreate(humi_sensor_task, "humi_task", 2048, NULL, 5, NULL);
    xTaskCreate(press_sensor_task, "press_task", 2048, NULL, 5, NULL);
    xTaskCreate(data_process_task, "process_task", 4096, NULL, 6, NULL);
}
```

### 示例3：按键触发异步处理

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/event_groups.h"
#include "driver/gpio.h"
#include "esp_event.h"
#include "esp_log.h"

static const char *TAG = "BUTTON_EVENT";

#define BUTTON_GPIO         GPIO_NUM_0

// 自定义事件
ESP_EVENT_DEFINE_BASE(BUTTON_EVENT_BASE);

enum {
    BUTTON_EVENT_PRESSED,
    BUTTON_EVENT_RELEASED,
    BUTTON_EVENT_LONG_PRESS,
};

static EventGroupHandle_t button_event_group;
#define BUTTON_PRESSED_BIT  BIT0

// GPIO中断处理
static void IRAM_ATTR button_isr_handler(void *arg)
{
    BaseType_t hpw = pdFALSE;
    xEventGroupSetBitsFromISR(button_event_group, BUTTON_PRESSED_BIT, &hpw);
    
    // 发送事件到默认循环
    int level = gpio_get_level(BUTTON_GPIO);
    int event_id = (level == 0) ? BUTTON_EVENT_PRESSED : BUTTON_EVENT_RELEASED;
    
    esp_event_isr_post(BUTTON_EVENT_BASE, event_id, NULL, 0, NULL);
    
    portYIELD_FROM_ISR(hpw);
}

// 事件处理器
void button_event_handler(void *handler_args, esp_event_base_t base, 
                          int32_t id, void *event_data)
{
    if (base == BUTTON_EVENT_BASE) {
        switch (id) {
            case BUTTON_EVENT_PRESSED:
                ESP_LOGI(TAG, "Button pressed");
                break;
            case BUTTON_EVENT_RELEASED:
                ESP_LOGI(TAG, "Button released");
                break;
            case BUTTON_EVENT_LONG_PRESS:
                ESP_LOGI(TAG, "Button long press");
                break;
        }
    }
}

// 按键检测任务（消抖+长按检测）
void button_task(void *arg)
{
    uint32_t press_time = 0;
    bool pressed = false;
    
    while (1) {
        EventBits_t bits = xEventGroupWaitBits(
            button_event_group,
            BUTTON_PRESSED_BIT,
            pdTRUE, pdFALSE, pdMS_TO_TICKS(100));
        
        if (bits & BUTTON_PRESSED_BIT) {
            vTaskDelay(pdMS_TO_TICKS(20));  // 消抖
            
            int level = gpio_get_level(BUTTON_GPIO);
            
            if (level == 0 && !pressed) {
                // 按下
                pressed = true;
                press_time = xTaskGetTickCount();
                ESP_LOGI(TAG, "Button debounced press");
                
            } else if (level == 1 && pressed) {
                // 释放
                uint32_t duration = (xTaskGetTickCount() - press_time) * portTICK_PERIOD_MS;
                pressed = false;
                
                if (duration > 2000) {
                    esp_event_post(BUTTON_EVENT_BASE, BUTTON_EVENT_LONG_PRESS, 
                                   &duration, sizeof(duration), portMAX_DELAY);
                }
                
                ESP_LOGI(TAG, "Button released after %lu ms", duration);
            }
        }
    }
}

void app_main(void)
{
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    
    // 注册事件处理器
    ESP_ERROR_CHECK(esp_event_handler_register(
        BUTTON_EVENT_BASE, ESP_EVENT_ANY_ID, &button_event_handler, NULL));
    
    button_event_group = xEventGroupCreate();
    
    // 配置GPIO
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << BUTTON_GPIO),
        .mode = GPIO_MODE_INPUT,
        .pull_up_en = GPIO_PULLUP_ENABLE,
        .intr_type = GPIO_INTR_ANYEDGE,
    };
    gpio_config(&io_conf);
    
    gpio_install_isr_service(ESP_INTR_FLAG_DEFAULT);
    gpio_isr_handler_add(BUTTON_GPIO, button_isr_handler, NULL);
    
    xTaskCreate(button_task, "button_task", 2048, NULL, 5, NULL);
    
    ESP_LOGI(TAG, "Button event system started");
}
```
