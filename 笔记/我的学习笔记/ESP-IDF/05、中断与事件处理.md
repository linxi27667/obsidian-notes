# ESP-IDF 中断与事件处理 （异步事件响应与中断管理）

## 核心概念

- **中断(Interrupt)** - 硬件或软件触发的异步事件，打断当前程序执行
- **ISR** - 中断服务程序(Interrupt Service Routine)，要求快速、简洁
- **中断延迟** - 从中断触发到ISR开始执行的时间
- **临界区** - 需要保护不受中断干扰的代码段

---

## 一、中断基础

### 1.1 中断类型

```
ESP32中断架构：
┌─────────────────────────────────────────────────────────┐
│                      CPU中断系统                         │
├─────────────────────────────────────────────────────────┤
│  ┌─────────────┐  ┌─────────────┐  ┌─────────────────┐ │
│  │  外部中断    │  │  定时器中断  │  │   软件中断      │ │
│  │  (GPIO)     │  │  (Timer)    │  │  (Software)     │ │
│  └─────────────┘  └─────────────┘  └─────────────────┘ │
├─────────────────────────────────────────────────────────┤
│  ┌─────────────┐  ┌─────────────┐  ┌─────────────────┐ │
│  │  UART中断   │  │   SPI中断   │  │   I2C中断       │ │
│  │  (串口)     │  │  (总线)     │  │  (总线)         │ │
│  └─────────────┘  └─────────────┘  └─────────────────┘ │
├─────────────────────────────────────────────────────────┤
│  ┌─────────────┐  ┌─────────────┐  ┌─────────────────┐ │
│  │  WiFi中断   │  │   蓝牙中断  │  │   看门狗中断    │ │
│  │  (无线)     │  │  (BLE)      │  │  (WDT)          │ │
│  └─────────────┘  └─────────────┘  └─────────────────┘ │
└─────────────────────────────────────────────────────────┘
```

| 中断类型 | 优先级 | 说明 |
|----------|--------|------|
| **不可屏蔽中断** | 最高 | NMI，如看门狗 |
| **高优先级中断** | Level 5-6 | 关键外设 |
| **中优先级中断** | Level 3-4 | 一般外设 |
| **低优先级中断** | Level 1-2 | 非关键外设 |

### 1.2 中断注册流程

```c
#include "esp_intr_alloc.h"

// 1. 配置GPIO中断（以GPIO为例）
gpio_config_t io_conf = {
    .pin_bit_mask = (1ULL << GPIO_NUM_0),
    .mode = GPIO_MODE_INPUT,
    .pull_up_en = GPIO_PULLUP_ENABLE,
    .intr_type = GPIO_INTR_NEGEDGE  // 下降沿触发
};
gpio_config(&io_conf);

// 2. 安装GPIO ISR服务
gpio_install_isr_service(ESP_INTR_FLAG_DEFAULT);

// 3. 添加中断处理函数
gpio_isr_handler_add(GPIO_NUM_0, gpio_isr_handler, (void *)GPIO_NUM_0);
```

## 二、GPIO中断配置

### 2.1 触发条件配置

```c
// 配置不同触发方式
gpio_config_t io_conf = {
    .pin_bit_mask = (1ULL << GPIO_NUM_0),
    .mode = GPIO_MODE_INPUT,
    .pull_up_en = GPIO_PULLUP_ENABLE,
};

// 上升沿触发（信号从低到高）
io_conf.intr_type = GPIO_INTR_POSEDGE;
gpio_config(&io_conf);

// 下降沿触发（信号从高到低）
io_conf.intr_type = GPIO_INTR_NEGEDGE;
gpio_config(&io_conf);

// 任意边沿触发
gpio_config_t io_conf.intr_type = GPIO_INTR_ANYEDGE;
gpio_config(&io_conf);

// 低电平触发（持续触发）
io_conf.intr_type = GPIO_INTR_LOW_LEVEL;
gpio_config(&io_conf);
```

**触发方式对比：**

```
信号波形:         上升沿          下降沿
        ─────┐         ┌─────────┐
             │         │         │
             └─────────┘         └─────
             ↑                  ↑
             posedge            negedge

边沿触发: 只在跳变时触发一次
电平触发: 只要满足电平条件就持续触发
```

### 2.2 ISR处理函数

```c
#include "freertos/FreeRTOS.h"
#include "freertos/queue.h"

static QueueHandle_t gpio_evt_queue = NULL;

// ISR函数 - 必须在IRAM中执行（加IRAM_ATTR）
static void IRAM_ATTR gpio_isr_handler(void *arg)
{
    uint32_t gpio_num = (uint32_t)arg;
    
    // ISR中只执行最小操作
    // 1. 发送队列通知任务
    xQueueSendFromISR(gpio_evt_queue, &gpio_num, NULL);
    
    // 2. 或设置标志位
    // gpio_intr_flag = true;
    
    // 3. 或释放信号量
    // xSemaphoreGiveFromISR(xSemaphore, NULL);
}
```

**ISR编程规范：**

| 允许操作 | 禁止操作 |
|----------|----------|
| `xQueueSendFromISR()` | `printf()` / `ESP_LOGx()` |
| `xSemaphoreGiveFromISR()` | `vTaskDelay()` |
| 设置标志位 | 复杂计算 |
| 清除中断标志 | 访问可能阻塞的资源 |

## 三、中断服务程序设计

### 3.1 快速响应设计

```c
// 推荐：ISR快速响应 + 任务处理
static volatile bool data_ready = false;
static uint32_t interrupt_count = 0;

static void IRAM_ATTR quick_isr_handler(void *arg)
{
    // 极简ISR：只设置标志
    data_ready = true;
    interrupt_count++;
}

// 任务中处理数据
void data_process_task(void *arg)
{
    while (1) {
        if (data_ready) {
            data_ready = false;
            
            // 在这里进行复杂处理
            process_sensor_data();
            ESP_LOGI(TAG, "Processed interrupt #%lu", interrupt_count);
        }
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}
```

---

### 3.2 延迟处理模式

```c
// 使用队列进行延迟处理
#define GPIO_QUEUE_SIZE 10

static QueueHandle_t gpio_queue = NULL;

typedef struct {
    uint32_t gpio_num;
    uint64_t timestamp;
} gpio_event_t;

static void IRAM_ATTR gpio_isr_handler(void *arg)
{
    gpio_event_t evt;
    evt.gpio_num = (uint32_t)arg;
    evt.timestamp = esp_timer_get_time();
    
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    xQueueSendFromISR(gpio_queue, &evt, &xHigherPriorityTaskWoken);
    
    if (xHigherPriorityTaskWoken) {
        portYIELD_FROM_ISR();
    }
}

void gpio_event_task(void *arg)
{
    gpio_event_t evt;
    
    while (1) {
        if (xQueueReceive(gpio_queue, &evt, portMAX_DELAY)) {
            ESP_LOGI(TAG, "GPIO %lu triggered at %llu us", 
                     evt.gpio_num, evt.timestamp);
            
            // 处理事件
            handle_gpio_event(evt.gpio_num);
        }
    }
}
```

## 四、中断与任务通信

### 4.1 使用队列通信

```c
// 创建队列
gpio_queue = xQueueCreate(10, sizeof(uint32_t));

// ISR发送
static void IRAM_ATTR isr_handler(void *arg)
{
    uint32_t gpio = (uint32_t)arg;
    BaseType_t hpw = pdFALSE;
    xQueueSendFromISR(gpio_queue, &gpio, &hpw);
    portYIELD_FROM_ISR(hpw);
}

// 任务接收
void task_handler(void *arg)
{
    uint32_t gpio_num;
    while (xQueueReceive(gpio_queue, &gpio_num, portMAX_DELAY)) {
        // 处理
    }
}
```

---

### 4.2 使用信号量通信

```c
#include "freertos/semphr.h"

static SemaphoreHandle_t xBinarySemaphore = NULL;

// ISR释放信号量
static void IRAM_ATTR isr_handler(void *arg)
{
    BaseType_t hpw = pdFALSE;
    xSemaphoreGiveFromISR(xBinarySemaphore, &hpw);
    portYIELD_FROM_ISR(hpw);
}

// 任务等待信号量
void sync_task(void *arg)
{
    xBinarySemaphore = xSemaphoreCreateBinary();
    
    while (1) {
        if (xSemaphoreTake(xBinarySemaphore, portMAX_DELAY) == pdTRUE) {
            ESP_LOGI(TAG, "Synchronized with interrupt");
        }
    }
}
```

---

### 4.3 使用任务通知（最轻量）

```c
static TaskHandle_t xHandlerTask = NULL;

// ISR发送通知
static void IRAM_ATTR isr_handler(void *arg)
{
    BaseType_t hpw = pdFALSE;
    vTaskNotifyGiveFromISR(xHandlerTask, &hpw);
    portYIELD_FROM_ISR(hpw);
}

// 任务等待通知
void notify_task(void *arg)
{
    xHandlerTask = xTaskGetCurrentTaskHandle();
    
    while (1) {
        ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
        ESP_LOGI(TAG, "Received notification");
    }
}
```

**通信方式对比：**

```
┌────────────────┬──────────┬────────────┬────────────────┐
│    方式        │  开销    │  数据传递  │   适用场景     │
├────────────────┼──────────┼────────────┼────────────────┤
│ 全局变量+标志   │ 最低     │ 无         │ 简单同步       │
│ 任务通知        │ 低       │ 32位值     │ 轻量级通知     │
│ 二值信号量      │ 中       │ 无         │ 单次同步       │
│ 队列            │ 较高     │ 任意数据   │ 传递数据/缓冲  │
└────────────────┴──────────┴────────────┴────────────────┘
```

## 五、临界区保护

### 5.1 禁用中断

```c
#include "freertos/task.h"

// 进入临界区（禁用中断）
portENTER_CRITICAL(&spinlock);

// 受保护的代码
shared_counter++;

// 退出临界区（恢复中断）
portEXIT_CRITICAL(&spinlock);
```

---

### 5.2 使用互斥锁

```c
#include "freertos/semphr.h"

static SemaphoreHandle_t xMutex = NULL;

void protected_function(void)
{
    if (xSemaphoreTake(xMutex, portMAX_DELAY) == pdTRUE) {
        // 临界区
        modify_shared_resource();
        
        xSemaphoreGive(xMutex);
    }
}
```

---

### 5.3 自旋锁（多核保护）

```c
#include "freertos/task.h"

static portMUX_TYPE spinlock = portMUX_INITIALIZER_UNLOCKED;

void multi_core_safe_function(void)
{
    portENTER_CRITICAL(&spinlock);
    
    // 受保护的共享资源访问
    shared_data.value++;
    shared_data.timestamp = esp_timer_get_time();
    
    portEXIT_CRITICAL(&spinlock);
}
```

**保护机制对比：**

| 机制 | 保护范围 | 延迟影响 | 适用场景 |
|------|----------|----------|----------|
| `portENTER_CRITICAL` | 单核中断 | 禁用中断 | 短时间操作 |
| `portMUX` | 多核+中断 | 忙等待 | 跨核共享数据 |
| `Mutex` | 任务间 | 任务切换 | 长时间操作 |

## 六、中断延迟与优化

### 6.1 延迟来源

```
中断响应延迟组成：

触发 ──> 硬件响应 ──> 上下文保存 ──> ISR执行
 │          │             │            │
 │         ~12ns        ~50-100ns    用户代码
 │        (硬件固定)    (RTOS开销)
 │
└─ 总延迟：通常 < 1us (不考虑其他中断阻塞)
```

---

### 6.2 优化策略

```c
// 1. 使用IRAM_ATTR将ISR放在IRAM
static void IRAM_ATTR fast_isr_handler(void *arg)
{
    // 避免Flash访问延迟
}

// 2. 减少ISR执行时间
static void IRAM_ATTR optimized_isr(void *arg)
{
    // 只设置标志，不处理
    event_flag = true;
    // 不调用任何函数
}

// 3. 提高中断优先级（如果需要）
#define HIGH_PRIORITY_INTR  ESP_INTR_FLAG_LEVEL5

gpio_install_isr_service(HIGH_PRIORITY_INTR);
```

## 七、完整示例

### 示例1：编码器脉冲计数

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/queue.h"
#include "driver/gpio.h"
#include "driver/pulse_cnt.h"
#include "esp_log.h"

static const char *TAG = "ENCODER";

// 编码器引脚
#define ENCODER_A_GPIO      GPIO_NUM_4
#define ENCODER_B_GPIO      GPIO_NUM_5

// 使用PCNT硬件计数器
pcnt_unit_handle_t pcnt_unit = NULL;

void encoder_init(void)
{
    // 配置PCNT单元
    pcnt_unit_config_t unit_config = {
        .high_limit = 10000,
        .low_limit = -10000,
    };
    ESP_ERROR_CHECK(pcnt_new_unit(&unit_config, &pcnt_unit));
    
    // 配置通道A
    pcnt_chan_config_t chan_a_config = {
        .edge_gpio_num = ENCODER_A_GPIO,
        .level_gpio_num = ENCODER_B_GPIO,
    };
    pcnt_channel_handle_t pcnt_chan_a = NULL;
    ESP_ERROR_CHECK(pcnt_new_channel(pcnt_unit, &chan_a_config, &pcnt_chan_a));
    
    // 配置边沿动作
    pcnt_channel_set_edge_action(pcnt_chan_a,
                                  PCNT_CHANNEL_EDGE_ACTION_DECREASE,  // 下降沿
                                  PCNT_CHANNEL_EDGE_ACTION_INCREASE); // 上升沿
    pcnt_channel_set_level_action(pcnt_chan_a,
                                   PCNT_CHANNEL_LEVEL_ACTION_KEEP,    // 高电平
                                   PCNT_CHANNEL_LEVEL_ACTION_INVERSE); // 低电平
    
    // 启动计数器
    ESP_ERROR_CHECK(pcnt_unit_enable(pcnt_unit));
    ESP_ERROR_CHECK(pcnt_unit_clear_count(pcnt_unit));
    ESP_ERROR_CHECK(pcnt_unit_start(pcnt_unit));
    
    ESP_LOGI(TAG, "Encoder initialized on GPIO %d/%d", ENCODER_A_GPIO, ENCODER_B_GPIO);
}

void encoder_task(void *arg)
{
    int pulse_count = 0;
    int last_count = 0;
    
    while (1) {
        ESP_ERROR_CHECK(pcnt_unit_get_count(pcnt_unit, &pulse_count));
        
        if (pulse_count != last_count) {
            ESP_LOGI(TAG, "Encoder count: %d", pulse_count);
            last_count = pulse_count;
        }
        
        vTaskDelay(pdMS_TO_TICKS(100));
    }
}

void app_main(void)
{
    encoder_init();
    xTaskCreate(encoder_task, "encoder_task", 2048, NULL, 5, NULL);
}
```

### 示例2：超声波测距（输入捕获）

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "driver/gptimer.h"
#include "esp_log.h"

static const char *TAG = "ULTRASONIC";

// 超声波引脚
#define TRIG_GPIO       GPIO_NUM_5
#define ECHO_GPIO       GPIO_NUM_18

// GPTimer用于精确计时
gptimer_handle_t us_timer = NULL;

// 测量结果
static volatile uint32_t echo_start_time = 0;
static volatile uint32_t echo_end_time = 0;
static volatile bool measurement_done = false;

// Echo引脚中断处理
static void IRAM_ATTR echo_isr_handler(void *arg)
{
    uint64_t count;
    gptimer_get_raw_count(us_timer, &count);
    
    int level = gpio_get_level(ECHO_GPIO);
    
    if (level == 1) {
        // 上升沿 - 开始计时
        echo_start_time = (uint32_t)count;
    } else {
        // 下降沿 - 结束计时
        echo_end_time = (uint32_t)count;
        measurement_done = true;
    }
}

void ultrasonic_init(void)
{
    // 配置Trig引脚（输出）
    gpio_config_t trig_conf = {
        .pin_bit_mask = (1ULL << TRIG_GPIO),
        .mode = GPIO_MODE_OUTPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_DISABLE,
    };
    gpio_config(&trig_conf);
    gpio_set_level(TRIG_GPIO, 0);
    
    // 配置Echo引脚（输入+中断）
    gpio_config_t echo_conf = {
        .pin_bit_mask = (1ULL << ECHO_GPIO),
        .mode = GPIO_MODE_INPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_ANYEDGE,  // 任意边沿触发
    };
    gpio_config(&echo_conf);
    
    // 创建1MHz定时器（1us精度）
    gptimer_config_t timer_config = {
        .clk_src = GPTIMER_CLK_SRC_DEFAULT,
        .direction = GPTIMER_COUNT_UP,
        .resolution_hz = 1000000,  // 1MHz = 1us/tick
    };
    ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &us_timer));
    ESP_ERROR_CHECK(gptimer_enable(us_timer));
    ESP_ERROR_CHECK(gptimer_start(us_timer));
    
    // 安装GPIO中断
    gpio_install_isr_service(ESP_INTR_FLAG_DEFAULT);
    gpio_isr_handler_add(ECHO_GPIO, echo_isr_handler, NULL);
    
    ESP_LOGI(TAG, "Ultrasonic sensor initialized");
}

float measure_distance(void)
{
    measurement_done = false;
    
    // 发送10us触发脉冲
    gpio_set_level(TRIG_GPIO, 1);
    esp_rom_delay_us(10);
    gpio_set_level(TRIG_GPIO, 0);
    
    // 等待测量完成（超时30ms）
    int timeout = 300;
    while (!measurement_done && timeout-- > 0) {
        esp_rom_delay_us(100);
    }
    
    if (measurement_done) {
        // 计算距离：时间(us) * 声速(343m/s) / 2（往返）
        uint32_t duration = echo_end_time - echo_start_time;
        float distance = (duration * 0.0343f) / 2.0f;
        return distance;
    }
    
    return -1.0f;  // 超时
}

void ultrasonic_task(void *arg)
{
    while (1) {
        float distance = measure_distance();
        
        if (distance >= 0) {
            ESP_LOGI(TAG, "Distance: %.2f cm", distance);
        } else {
            ESP_LOGW(TAG, "Measurement timeout");
        }
        
        vTaskDelay(pdMS_TO_TICKS(500));
    }
}

void app_main(void)
{
    ultrasonic_init();
    xTaskCreate(ultrasonic_task, "ultrasonic", 2048, NULL, 5, NULL);
}
```

### 示例3：按键长按/短按识别

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "freertos/queue.h"
#include "driver/gpio.h"
#include "esp_timer.h"
#include "esp_log.h"

static const char *TAG = "BUTTON";

#define BUTTON_GPIO         GPIO_NUM_0
#define DEBOUNCE_MS         20
#define SHORT_PRESS_MS      500
#define LONG_PRESS_MS       2000

// 按键事件类型
typedef enum {
    BUTTON_PRESS,       // 按下
    BUTTON_RELEASE,     // 释放
    BUTTON_SHORT,       // 短按
    BUTTON_LONG,        // 长按
} button_event_t;

// 按键状态机
typedef enum {
    STATE_IDLE,         // 空闲
    STATE_PRESSED,      // 已按下
    STATE_SHORT,        // 短按确认
    STATE_LONG,         // 长按确认
} button_state_t;

typedef struct {
    button_event_t event;
    uint32_t duration_ms;
} button_msg_t;

static QueueHandle_t button_queue = NULL;
static button_state_t btn_state = STATE_IDLE;
static uint64_t press_time = 0;

static void IRAM_ATTR button_isr_handler(void *arg)
{
    uint32_t gpio_num = (uint32_t)arg;
    xQueueSendFromISR(button_queue, &gpio_num, NULL);
}

void button_task(void *arg)
{
    uint32_t gpio_num;
    button_msg_t msg;
    
    while (1) {
        if (xQueueReceive(button_queue, &gpio_num, pdMS_TO_TICKS(10))) {
            int level = gpio_get_level(BUTTON_GPIO);
            uint64_t current_time = esp_timer_get_time() / 1000;  // 转换为ms
            
            switch (btn_state) {
                case STATE_IDLE:
                    if (level == 0) {
                        // 按键按下
                        press_time = current_time;
                        btn_state = STATE_PRESSED;
                        
                        msg.event = BUTTON_PRESS;
                        msg.duration_ms = 0;
                        ESP_LOGI(TAG, "Button pressed");
                    }
                    break;
                    
                case STATE_PRESSED:
                case STATE_SHORT:
                    if (level == 1) {
                        // 按键释放
                        uint32_t duration = current_time - press_time;
                        
                        if (duration >= LONG_PRESS_MS) {
                            msg.event = BUTTON_LONG;
                        } else if (duration >= SHORT_PRESS_MS) {
                            msg.event = BUTTON_SHORT;
                        } else {
                            msg.event = BUTTON_RELEASE;
                        }
                        
                        msg.duration_ms = duration;
                        btn_state = STATE_IDLE;
                        
                        ESP_LOGI(TAG, "Button released after %lu ms (%s)",
                                duration,
                                msg.event == BUTTON_LONG ? "LONG" :
                                msg.event == BUTTON_SHORT ? "SHORT" : "QUICK");
                    } else {
                        // 检查长按
                        uint32_t duration = current_time - press_time;
                        if (duration >= LONG_PRESS_MS && btn_state != STATE_LONG) {
                            btn_state = STATE_LONG;
                            msg.event = BUTTON_LONG;
                            msg.duration_ms = duration;
                            ESP_LOGI(TAG, "Button LONG press detected");
                        } else if (duration >= SHORT_PRESS_MS && btn_state == STATE_PRESSED) {
                            btn_state = STATE_SHORT;
                            msg.event = BUTTON_SHORT;
                            msg.duration_ms = duration;
                            ESP_LOGI(TAG, "Button SHORT press detected");
                        }
                    }
                    break;
                    
                default:
                    btn_state = STATE_IDLE;
                    break;
            }
        }
    }
}

void button_init(void)
{
    // 配置GPIO
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << BUTTON_GPIO),
        .mode = GPIO_MODE_INPUT,
        .pull_up_en = GPIO_PULLUP_ENABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_ANYEDGE,  // 任意边沿触发
    };
    gpio_config(&io_conf);
    
    // 创建队列
    button_queue = xQueueCreate(10, sizeof(uint32_t));
    
    // 安装中断
    gpio_install_isr_service(ESP_INTR_FLAG_DEFAULT);
    gpio_isr_handler_add(BUTTON_GPIO, button_isr_handler, (void *)BUTTON_GPIO);
    
    ESP_LOGI(TAG, "Button initialized on GPIO %d", BUTTON_GPIO);
}

void app_main(void)
{
    button_init();
    xTaskCreate(button_task, "button_task", 2048, NULL, 10, NULL);
    
    ESP_LOGI(TAG, "Long press: >%d ms, Short press: %d-%d ms",
             LONG_PRESS_MS, SHORT_PRESS_MS, LONG_PRESS_MS);
}
```

## 附录：中断API速查表

### GPIO中断

| API | 说明 |
|-----|------|
| `gpio_install_isr_service()` | 安装GPIO ISR服务 |
| `gpio_uninstall_isr_service()` | 卸载ISR服务 |
| `gpio_isr_handler_add()` | 添加中断处理 |
| `gpio_isr_handler_remove()` | 移除中断处理 |
| `gpio_wakeup_enable()` | 启用GPIO唤醒 |

### 中断标志

| 标志 | 说明 |
|------|------|
| `ESP_INTR_FLAG_DEFAULT` | 默认标志（Level 1-3） |
| `ESP_INTR_FLAG_LEVEL1-6` | 指定中断优先级 |
| `ESP_INTR_FLAG_NMI` | 不可屏蔽中断 |
| `ESP_INTR_FLAG_SHARED` | 共享中断 |
| `ESP_INTR_FLAG_IRAM` | ISR位于IRAM |

> ⚠️ **v5.3.5+ SPI中断限制**：从v5.3.5开始，SPI master/slave 初始化不再接受 `ESP_INTR_FLAG_SHARED` 作为中断标志。

### ISR安全API

| API | 说明 |
|-----|------|
| `xQueueSendFromISR()` | 队列发送 |
| `xQueueReceiveFromISR()` | 队列接收 |
| `xSemaphoreGiveFromISR()` | 释放信号量 |
| `xTaskNotifyGiveFromISR()` | 任务通知 |
| `vTaskNotifyGiveFromISR()` | 任务通知（另一种形式） |
| `portYIELD_FROM_ISR()` | 触发上下文切换 |
