# ESP-IDF 定时器与延时 （系统时间管理与硬件定时器）

## 核心概念

- **FreeRTOS延时** - 基于tick的软件延时，适合一般任务调度
- **硬件定时器** - GPTimer提供更高精度和独立于任务的定时能力
- **esp_timer** - 高分辨率软件定时器，基于系统时间
- **看门狗定时器** - 防止任务卡死，保证系统稳定性

---

## 一、系统延时函数

### 1.1 相对延时 vTaskDelay

```c
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"

void task_function(void *pvParameters)
{
    while (1) {
        // 执行操作
        do_something();
        
        // 延时1000ms（从当前时间点开始算）
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}
```

| 宏 | 说明 |
|----|------|
| `pdMS_TO_TICKS(ms)` | 毫秒转换为tick数 |
| `pdTICKS_TO_MS(ticks)` | tick数转换为毫秒 |
| `portTICK_PERIOD_MS` | 每个tick的毫秒数（默认10ms） |

---

### 1.2 绝对延时 vTaskDelayUntil

```c
void periodic_task(void *pvParameters)
{
    TickType_t xLastWakeTime;
    const TickType_t xFrequency = pdMS_TO_TICKS(100);  // 100ms周期
    
    // 初始化上次唤醒时间
    xLastWakeTime = xTaskGetTickCount();
    
    while (1) {
        // 执行周期任务
        process_data();
        
        // 绝对延时，确保固定周期（不受任务执行时间影响）
        vTaskDelayUntil(&xLastWakeTime, xFrequency);
    }
}
```

**延时方式对比：**

```
vTaskDelay (相对延时):          vTaskDelayUntil (绝对延时):
                                
时间轴:                         时间轴:
├─执行─┤                        ├─执行─┤
       ├─────延时─────┤                ├─────┤
                        ├─执行─┤             ├─────┤
                               ├─────延时─────┤    ├─────┤
                                                 固定周期100ms
实际周期 = 执行时间 + 延时
```

## 二、GPTimer通用定时器

### 2.1 定时器初始化

```c
#include "driver/gptimer.h"

gptimer_handle_t gptimer = NULL;

// 定时器配置
gptimer_config_t timer_config = {
    .clk_src = GPTIMER_CLK_SRC_DEFAULT,    // 时钟源
    .direction = GPTIMER_COUNT_UP,          // 向上计数
    .resolution_hz = 1 * 1000 * 1000,       // 1MHz分辨率（1us/tick）
};
ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &gptimer));
```

| 参数 | 说明 | 常用值 |
|------|------|--------|
| `clk_src` | 时钟源 | `GPTIMER_CLK_SRC_DEFAULT` |
| `direction` | 计数方向 | `GPTIMER_COUNT_UP/DOWN` |
| `resolution_hz` | 分辨率 | 1MHz(1us), 10kHz(100us) |

---

### 2.2 报警配置与回调

```c
// 报警回调函数
static bool IRAM_ATTR gptimer_alarm_cb(gptimer_handle_t timer, 
                                       const gptimer_alarm_event_data_t *edata,
                                       void *user_ctx)
{
    // 在中断上下文中执行
    // 快速操作，避免复杂计算
    
    BaseType_t high_task_awoken = pdFALSE;
    
    // 通知任务
    // xSemaphoreGiveFromISR(...)
    
    return high_task_awoken == pdTRUE;
}

// 配置报警
gptimer_alarm_config_t alarm_config = {
    .alarm_count = 1000000,      // 1秒（1MHz时钟）
    .reload_count = 0,           // 重载值
    .flags.auto_reload_on_alarm = true,  // 自动重载
};
ESP_ERROR_CHECK(gptimer_set_alarm_action(gptimer, &alarm_config));

// 注册回调
gptimer_event_callbacks_t cbs = {
    .on_alarm = gptimer_alarm_cb,
};
ESP_ERROR_CHECK(gptimer_register_event_callbacks(gptimer, &cbs, NULL));

// 启动定时器
ESP_ERROR_CHECK(gptimer_enable(gptimer));
ESP_ERROR_CHECK(gptimer_start(gptimer));
```

### 4.2 esp_timer高精度定时器

```c
#include "esp_timer.h"

// 获取系统启动后的微秒数
int64_t us = esp_timer_get_time();
ESP_LOGI(TAG, "Time since boot: %lld us", us);
```

**时间源对比：**

| 函数 | 精度 | 范围 | 用途 |
|------|------|------|------|
| `xTaskGetTickCount()` | 10ms | 系统运行期间 | 任务延时 |
| `esp_timer_get_time()` | 1us | ~300年 | 精确计时 |
| `gettimeofday()` | 1us | Unix时间 | 实际时间 |

### 4.3 esp_timer软件定时器

```c
#include "esp_timer.h"

// 定时器回调
void timer_callback(void *arg)
{
    ESP_LOGI(TAG, "Timer triggered!");
}

// 创建定时器
const esp_timer_create_args_t timer_args = {
    .callback = &timer_callback,
    .arg = NULL,
    .dispatch_method = ESP_TIMER_TASK,      // 在定时器任务中执行
    .name = "my_timer"
};

esp_timer_handle_t timer;
ESP_ERROR_CHECK(esp_timer_create(&timer_args, &timer));

// 启动一次性定时器（100ms后触发）
ESP_ERROR_CHECK(esp_timer_start_once(timer, 100000));  // 单位：us

// 启动周期性定时器（每1秒触发）
ESP_ERROR_CHECK(esp_timer_start_periodic(timer, 1000000));

// 停止定时器
ESP_ERROR_CHECK(esp_timer_stop(timer));

// 删除定时器
ESP_ERROR_CHECK(esp_timer_delete(timer));
```

| 分发方式 | 说明 |
|----------|------|
| `ESP_TIMER_TASK` | 在专用定时器任务中执行（默认） |
| `ESP_TIMER_ISR` | 在中断中执行（更快，限制更多） |

## 五、看门狗定时器

### 5.1 任务看门狗

```c
#include "esp_task_wdt.h"

// 初始化任务看门狗（一般在app_main前自动初始化）
esp_task_wdt_config_t wdt_config = {
    .timeout_ms = 5000,          // 5秒超时
    .idle_core_mask = (1 << portNUM_PROCESSORS) - 1,  // 监控所有核心
    .trigger_panic = true        // 超时触发panic
};
ESP_ERROR_CHECK(esp_task_wdt_init(&wdt_config));

// 添加当前任务到看门狗监控
ESP_ERROR_CHECK(esp_task_wdt_add(NULL));

// 喂狗（在任务循环中定期调用）
while (1) {
    // 执行任务
    do_work();
    
    // 重置看门狗
    ESP_ERROR_CHECK(esp_task_wdt_reset());
    
    vTaskDelay(pdMS_TO_TICKS(100));
}

// 从看门狗移除任务
ESP_ERROR_CHECK(esp_task_wdt_delete(NULL));
```

```c
    }
}
```

```
┌─────────────────┬─────────────┬─────────────┬────────────────┐
```

## 七、完整示例

### 示例1：周期性任务调度

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gptimer.h"
#include "esp_log.h"

static const char *TAG = "PERIODIC";

#define TIMER_RESOLUTION_HZ 1000000  // 1MHz
#define ALARM_INTERVAL_US   100000   // 100ms

gptimer_handle_t periodic_timer = NULL;
static volatile uint32_t alarm_count = 0;

// 定时器回调
static bool IRAM_ATTR periodic_timer_cb(gptimer_handle_t timer,
                                        const gptimer_alarm_event_data_t *edata,
                                        void *user_ctx)
{
    alarm_count++;
    return false;
}

void periodic_task(void *arg)
{
    uint32_t last_count = 0;
    
    while (1) {
        if (alarm_count != last_count) {
            ESP_LOGI(TAG, "Alarm triggered %lu times", alarm_count);
            last_count = alarm_count;
            
            // 执行周期任务
            // 如：读取传感器、更新显示等
        }
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}

void app_main(void)
{
    // 配置定时器
    gptimer_config_t timer_config = {
        .clk_src = GPTIMER_CLK_SRC_DEFAULT,
        .direction = GPTIMER_COUNT_UP,
        .resolution_hz = TIMER_RESOLUTION_HZ,
    };
    ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &periodic_timer));
    
    // 配置报警
    gptimer_alarm_config_t alarm_config = {
        .alarm_count = ALARM_INTERVAL_US,
        .reload_count = 0,
        .flags.auto_reload_on_alarm = true,
    };
    ESP_ERROR_CHECK(gptimer_set_alarm_action(periodic_timer, &alarm_config));
    
    // 注册回调
    gptimer_event_callbacks_t cbs = {
        .on_alarm = periodic_timer_cb,
    };
    ESP_ERROR_CHECK(gptimer_register_event_callbacks(periodic_timer, &cbs, NULL));
    
    // 启动定时器
    ESP_ERROR_CHECK(gptimer_enable(periodic_timer));
    ESP_ERROR_CHECK(gptimer_start(periodic_timer));
    
    // 创建处理任务
    xTaskCreate(periodic_task, "periodic", 2048, NULL, 5, NULL);
    
    ESP_LOGI(TAG, "Periodic timer started: %d us interval", ALARM_INTERVAL_US);
}
```

### 示例2：软件PWM实现

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "driver/gptimer.h"
#include "driver/gpio.h"
#include "esp_log.h"

static const char *TAG = "SOFT_PWM";

#define PWM_GPIO        GPIO_NUM_2
#define PWM_FREQ_HZ     1000        // 1kHz PWM频率
#define PWM_RESOLUTION  100         // 分辨率100级

gptimer_handle_t pwm_timer = NULL;
static volatile uint32_t pwm_counter = 0;
static volatile uint32_t pwm_duty = 50;  // 占空比50%

// PWM定时器回调
static bool IRAM_ATTR pwm_timer_cb(gptimer_handle_t timer,
                                   const gptimer_alarm_event_data_t *edata,
                                   void *user_ctx)
{
    pwm_counter++;
    if (pwm_counter >= PWM_RESOLUTION) {
        pwm_counter = 0;
    }
    
    // 设置GPIO电平
    if (pwm_counter < pwm_duty) {
        gpio_set_level(PWM_GPIO, 1);
    } else {
        gpio_set_level(PWM_GPIO, 0);
    }
    
    return false;
}

void pwm_init(void)
{
    // 配置GPIO
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << PWM_GPIO),
        .mode = GPIO_MODE_OUTPUT,
        .pull_up_en = GPIO_PULLUP_DISABLE,
        .pull_down_en = GPIO_PULLDOWN_DISABLE,
        .intr_type = GPIO_INTR_DISABLE,
    };
    gpio_config(&io_conf);
    
    // 计算定时器参数
    uint32_t timer_period_us = 1000000 / (PWM_FREQ_HZ * PWM_RESOLUTION);
    
    // 配置定时器
    gptimer_config_t timer_config = {
        .clk_src = GPTIMER_CLK_SRC_DEFAULT,
        .direction = GPTIMER_COUNT_UP,
        .resolution_hz = 1000000,  // 1MHz = 1us/tick
    };
    ESP_ERROR_CHECK(gptimer_new_timer(&timer_config, &pwm_timer));
    
    // 配置报警（每个PWM tick触发一次）
    gptimer_alarm_config_t alarm_config = {
        .alarm_count = timer_period_us,
        .reload_count = 0,
        .flags.auto_reload_on_alarm = true,
    };
    ESP_ERROR_CHECK(gptimer_set_alarm_action(pwm_timer, &alarm_config));
    
    // 注册回调
    gptimer_event_callbacks_t cbs = {
        .on_alarm = pwm_timer_cb,
    };
    ESP_ERROR_CHECK(gptimer_register_event_callbacks(pwm_timer, &cbs, NULL));
    
    // 启动
    ESP_ERROR_CHECK(gptimer_enable(pwm_timer));
    ESP_ERROR_CHECK(gptimer_start(pwm_timer));
    
    ESP_LOGI(TAG, "Software PWM: %d Hz, resolution=%d", PWM_FREQ_HZ, PWM_RESOLUTION);
}

void pwm_set_duty(uint32_t duty)
{
    if (duty > PWM_RESOLUTION) {
        duty = PWM_RESOLUTION;
    }
    pwm_duty = duty;
    ESP_LOGI(TAG, "PWM duty set to %lu%%", (duty * 100) / PWM_RESOLUTION);
}

void app_main(void)
{
    pwm_init();
    
    // 演示呼吸灯效果
    int direction = 1;
    while (1) {
        pwm_set_duty(pwm_duty);
        
        pwm_duty += direction * 5;
        if (pwm_duty >= PWM_RESOLUTION) {
            pwm_duty = PWM_RESOLUTION;
            direction = -1;
        } else if (pwm_duty <= 0) {
            pwm_duty = 0;
            direction = 1;
        }
        
        vTaskDelay(pdMS_TO_TICKS(50));
    }
}
```

### 示例3：超时检测机制

```c
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_timer.h"
#include "esp_log.h"

static const char *TAG = "TIMEOUT";

#define OPERATION_TIMEOUT_MS    5000    // 5秒超时

// 操作状态
typedef enum {
    OP_IDLE,
    OP_RUNNING,
    OP_COMPLETED,
    OP_TIMEOUT
} OperationState_t;

static volatile OperationState_t op_state = OP_IDLE;
static esp_timer_handle_t timeout_timer = NULL;

// 超时回调
void timeout_callback(void *arg)
{
    if (op_state == OP_RUNNING) {
        op_state = OP_TIMEOUT;
        ESP_LOGW(TAG, "Operation timeout!");
    }
}

// 模拟耗时操作
bool perform_operation_with_timeout(void)
{
    // 初始化超时定时器
    const esp_timer_create_args_t timer_args = {
        .callback = &timeout_callback,
        .arg = NULL,
        .dispatch_method = ESP_TIMER_TASK,
        .name = "timeout_timer"
    };
    
    if (timeout_timer == NULL) {
        ESP_ERROR_CHECK(esp_timer_create(&timer_args, &timeout_timer));
    }
    
    // 开始操作
    op_state = OP_RUNNING;
    ESP_LOGI(TAG, "Starting operation...");
    
    // 启动超时定时器
    ESP_ERROR_CHECK(esp_timer_start_once(timeout_timer, OPERATION_TIMEOUT_MS * 1000));
    
    // 模拟操作（可能耗时不定）
    int steps = 0;
    while (op_state == OP_RUNNING) {
        // 模拟工作
        vTaskDelay(pdMS_TO_TICKS(500));
        steps++;
        ESP_LOGI(TAG, "Operation step %d", steps);
        
        // 模拟随机完成（这里用3步完成）
        if (steps >= 3) {
            op_state = OP_COMPLETED;
        }
    }
    
    // 停止超时定时器
    esp_timer_stop(timeout_timer);
    
    if (op_state == OP_COMPLETED) {
        ESP_LOGI(TAG, "Operation completed successfully");
        return true;
    } else {
        ESP_LOGE(TAG, "Operation failed: timeout");
        return false;
    }
}

void app_main(void)
{
    // 执行带超时的操作
    bool result = perform_operation_with_timeout();
    
    ESP_LOGI(TAG, "Result: %s", result ? "SUCCESS" : "FAILED");
    
    // 清理
    if (timeout_timer != NULL) {
        esp_timer_delete(timeout_timer);
    }
}
```
